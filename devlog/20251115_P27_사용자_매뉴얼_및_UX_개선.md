```python
    rounded_r = round(r, 3)
    if member in highlighted_members:
        color = 'red'  # 하이라이트
    else:
        # 해당 동심원의 색상 사용
        color = radius_to_color.get(rounded_r, circle_colors_concentric[0])

    ax.plot(x, y, 'o', color=color, markersize=8, ...)
```

**색상 스키마**:
- 무지개 색상 사용 (`plt.cm.rainbow`)
- 안쪽 원(높은 global_sim): 보라색
- 중간 원: 파란색 → 초록색 → 노란색
- 바깥쪽 원(낮은 global_sim): 빨간색

**효과**:
- 각 area가 어느 global_sim 레벨에 속하는지 색상으로 즉시 파악 가능
- 시각적으로 더 풍부하고 정보가 많은 시각화
- 동심원과 area point의 일관된 색상 체계

```python
if not np.allclose(arr, arr.T, rtol=1e-5, atol=1e-8):
    diff = np.abs(arr - arr.T)
    max_diff_idx = np.unravel_index(np.argmax(diff), diff.shape)
    i, j = max_diff_idx
    return False, f"Matrix is not symmetric: matrix[{i},{j}]={arr[i,j]:.6f} but matrix[{j},{i}]={arr[j,i]:.6f}"
```

```python
if not np.allclose(diag, 1.0, rtol=1e-5, atol=1e-8):
```
